SELECT p.name, receiver.name, p.phone_number, p.license_plate, bsl.activity AS "Activity", origin.full_name AS "Origin", destination.full_name AS "Destination", atm.transaction_type
SELECT p.name, receiver.name, destination.full_name AS "Destination", flight.*
FROM people AS p
JOIN phone_calls AS caller ON p.phone_number=caller.caller
JOIN people AS receiver ON caller.receiver=receiver.phone_number
JOIN bakery_security_logs AS bsl ON p.license_plate=bsl.license_plate
JOIN bank_accounts AS ba ON p.id=ba.person_id
JOIN atm_transactions AS atm ON ba.account_number=atm.account_number
JOIN passengers AS pass ON pass.passport_number=p.passport_number
JOIN flights AS flight ON flight.id=pass.flight_id
JOIN airports AS origin ON origin.id=flight.origin_airport_id
JOIN airports AS destination ON destination.id=flight.destination_airport_id
WHERE
    bsl.year=2023
    AND bsl.month=7
    AND bsl.day=28
    AND caller.year=2023
    AND caller.month=7
    AND caller.day=28
    AND caller.duration<60
    AND atm.year=2023
    AND atm.month=7
    AND atm.day=28
    AND atm.atm_location="Leggett Street"
    AND flight.year=2023
    AND flight.month=7
    AND flight.day=29
    AND bsl.minute>=5
    AND bsl.minute<=25
    AND bsl.hour=10
    AND bsl.activity="exit"
ORDER BY flight.hour LIMIT 1
;
